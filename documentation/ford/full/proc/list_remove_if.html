list_remove_if

private subroutine list_remove_if(this, pred)

Remove elements fulfilling condition Removes from the container all the elements for which Predicate pred returns true. This calls the destructor of these objects and reduces the container size by the number of elements removed. The function calls pred(i%node%element) for each element (where i is an iterator to that element). Any of the elements in the list for which this returns true, are removed from the container.

subroutine list_remove_if( this, pred )

!> The list
  class(t_xfunit_assertion_list_ftl), intent(inout) :: this

!> THe predicate to use for removal selection
  procedure(predicate) :: pred

! Local variables
  type(t_xfunit_assertion_list_ftl_iterator) :: it, itnext

! Check that there are elements in the list
  if( associated(this%first) ) then

!   Initialise iteration
    it = this%begin()

!   Loop on the iterator range
    do while( it%associated() )

!     Preserve next in case that current is deleted
      itnext = it%next()

!     Check and remove
      if( pred( it%node%element ) ) then
        call this%erase(it)
      end if

!     Iterate
      it = itnext

    end do

  end if

end subroutine list_remove_if
