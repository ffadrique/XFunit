elemental function xfunit_unit_is_passed( ut ) result(res)

!> The unit test case
  class(t_xfunit_unit), intent(in) :: ut

!> The passed status
  logical :: res

! Compute the result
  res = ut%executed .and. ut%status == 0 .and. ( ut%failed_assertions == 0 )

end function xfunit_unit_is_passed
